#!/bin/bash
#
# Claude Code Statusline Script
#
# 출력 예시:
#   [Opus] 📁 moyo-frontend-backup | 🌿 main
#   ████░░░░░░ 42% | $0.12 | ⏱️ 5m 25s
#
# 설정 방법:
#   1. jq 설치 필요 (brew install jq)
#   2. ~/.claude/settings.json에 추가:
#      { "statusLine": { "type": "command", "command": "~/.claude/statusline-command.sh" } }
#
# 동작:
#   - Claude Code가 stdin으로 JSON 세션 데이터를 전달하면 파싱하여 2줄로 출력
#   - 1줄: 모델명, 디렉토리, git 브랜치
#   - 2줄: 컨텍스트 사용량 프로그레스바(임계값 색상), 세션 비용, 경과 시간
#   - API 토큰 소비 없음 (로컬 실행)
#

# jq 없으면 설치 안내 후 종료 (macOS: brew install jq / Windows: winget install jqlang.jq)
if ! command -v jq &>/dev/null; then
    echo "jq not found — install: brew install jq (macOS) / winget install jqlang.jq (Windows)"
    exit 0
fi

# Claude Code가 stdin으로 보내는 JSON 세션 데이터 읽기
input=$(cat)

# jq 1회 호출로 필요한 필드 전부 추출
# - @tsv 대신 join("\t") 사용: read가 연속 탭을 하나로 취급하는 문제 방지
# - used_percentage가 null이면 "NULL" 센티넬로 치환 (첫 API 응답 전 null)
IFS=$'\t' read -r MODEL DIR COST PCT DURATION_MS <<< "$(echo "$input" | jq -r '[
    (.model.display_name // "Unknown"),
    (.workspace.current_dir // ""),
    (.cost.total_cost_usd // 0 | tostring),
    (if .context_window.used_percentage == null then "NULL" else (.context_window.used_percentage | tostring) end),
    (.cost.total_duration_ms // 0 | tostring)
] | join("\t")')"
[ "$PCT" = "NULL" ] && PCT=""

# ANSI 색상 코드
CYAN='\033[36m'; GREEN='\033[32m'; YELLOW='\033[33m'; RED='\033[31m'; DIM='\033[2m'; RESET='\033[0m'

# Git 브랜치 (세션당 1회 조회, 이후 캐시 파일에서 읽기)
# - 세션 중 브랜치가 바뀔 일은 거의 없으므로 TTL 없이 캐시
# - 캐시 파일이 없을 때만 git 실행
CACHE_DIR="${TMPDIR:-/tmp}"
DIR_HASH=$(printf '%s' "$DIR" | cksum | cut -d' ' -f1)
CACHE_FILE="${CACHE_DIR}/statusline-git-cache-${DIR_HASH}"

if [ -f "$CACHE_FILE" ]; then
    BRANCH=$(cat "$CACHE_FILE")
else
    BRANCH=$(git -C "$DIR" branch --show-current 2>/dev/null || true)
    echo "$BRANCH" > "$CACHE_FILE" 2>/dev/null
fi

# 1줄: [모델명] 📁 디렉토리 | 🌿 브랜치
BRANCH_PART=""
[ -n "$BRANCH" ] && BRANCH_PART=" | 🌿 ${BRANCH}"
echo -e "${CYAN}[${MODEL}]${RESET} ${DIM}📁 ${DIR##*/}${BRANCH_PART}${RESET}"

# 2줄: 프로그레스바 퍼센트 | 비용 | 경과시간
# - 컨텍스트 사용률에 따라 바 색상 변경: <70% 녹색, 70-89% 노란색, 90%+ 빨간색
# - PCT가 빈 문자열이면 첫 API 응답 전이므로 대시 표시
if [ -n "$PCT" ]; then
    PCT_INT=$(printf "%.0f" "$PCT")

    if [ "$PCT_INT" -ge 90 ]; then BAR_COLOR="$RED"
    elif [ "$PCT_INT" -ge 70 ]; then BAR_COLOR="$YELLOW"
    else BAR_COLOR="$GREEN"; fi

    # 10칸 프로그레스바: █(사용) ░(남은)
    FILLED=$((PCT_INT / 10))
    EMPTY=$((10 - FILLED))
    BAR=""
    [ "$FILLED" -gt 0 ] && BAR=$(printf "%${FILLED}s" | tr ' ' '█')
    [ "$EMPTY" -gt 0 ] && BAR="${BAR}$(printf "%${EMPTY}s" | tr ' ' '░')"

    COST_FMT=$(printf '$%.2f' "$COST")
    MINS=$((DURATION_MS / 60000))
    SECS=$(((DURATION_MS % 60000) / 1000))

    echo -e "${BAR_COLOR}${BAR}${RESET} ${DIM}${PCT_INT}%${RESET} | ${YELLOW}${COST_FMT}${RESET} | ${DIM}⏱️ ${MINS}m ${SECS}s${RESET}"
else
    echo -e "${DIM}░░░░░░░░░░ --% | \$0.00 | ⏱️ 0m 0s${RESET}"
fi
